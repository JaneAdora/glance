# `tasks` Implementation Plan

> **For agentic workers:** REQUIRED SUB-SKILL: superpowers:executing-plans. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** Build the glance panel + standalone `tasks` widget per spec `docs/superpowers/specs/2026-05-25-tasks-design.md`.

**Architecture:** Shared `TasksCore` (in `src/tasks/`) wrapped by `TasksPanel` (`src/panels/tasks.rs`) and standalone `src/bin/tasks.rs`. Store coordinates with Claude Code via `fs2::FileExt::lock_exclusive` on the persistent `.lock`; writes are atomic via tmp + rename. Session resolver reads `cwd` from `~/.claude/projects/<slug>/<sid>.jsonl` (scans first 20 lines, since line 1 is `last-prompt`).

**Tech Stack:** Rust, ratatui 0.29, crossterm 0.28, serde/serde_json, toml 0.8, jiff 0.2, dirs 5, **fs2 0.4 (NEW dep)**.

use crate::tasks::task::{ClaudeTask, SessionId, TaskId};
use fs2::FileExt;
use std::fs::{self, File, OpenOptions};
use std::io::Write;
use std::path::{Path, PathBuf};
use std::time::{Duration, Instant, SystemTime};

#[derive(Debug)]
pub enum StoreError {
    Io(std::io::Error),
    LockTimeout,
    Serde(serde_json::Error),
}
impl From<std::io::Error> for StoreError { fn from(e: std::io::Error) -> Self { Self::Io(e) } }
impl From<serde_json::Error> for StoreError { fn from(e: serde_json::Error) -> Self { Self::Serde(e) } }

pub struct LoadedSession {
    pub session_id: SessionId,
    pub mtime: SystemTime,
    pub tasks: Vec<ClaudeTask>,
}

pub struct LoadResult {
    pub sessions: Vec<LoadedSession>,
    pub toasts: Vec<String>,
}

pub fn tasks_root() -> PathBuf {
    dirs::home_dir().unwrap().join(".claude/tasks")
}

pub struct LockGuard { _file: File }
// Drop releases the flock automatically when the File closes.

pub fn acquire_lock(session_dir: &Path) -> Result<LockGuard, StoreError> {
    let lock_path = session_dir.join(".lock");
    let file = OpenOptions::new()
        .create(true).read(true).write(true)
        .open(&lock_path)?;
    let deadline = Instant::now() + Duration::from_secs(1);
    loop {
        match file.try_lock_exclusive() {
            Ok(()) => return Ok(LockGuard { _file: file }),
            Err(_) if Instant::now() < deadline => {
                std::thread::sleep(Duration::from_millis(50));
            }
            Err(_) => return Err(StoreError::LockTimeout),
        }
    }
}

pub fn write_task(session_dir: &Path, task: &ClaudeTask) -> Result<(), StoreError> {
    fs::create_dir_all(session_dir)?;
    let _guard = acquire_lock(session_dir)?;
    let final_path = session_dir.join(format!("{}.json", task.id));
    let tmp_path = session_dir.join(format!("{}.json.tmp", task.id));
    let json = serde_json::to_vec_pretty(task)?;
    {
        let mut f = OpenOptions::new().create(true).write(true).truncate(true).open(&tmp_path)?;
        f.write_all(&json)?;
        f.sync_data()?;
    }
    fs::rename(&tmp_path, &final_path)?;
    Ok(())
}

pub fn delete_task(session_dir: &Path, id: &TaskId) -> Result<(), StoreError> {
    let _guard = acquire_lock(session_dir)?;
    let path = session_dir.join(format!("{}.json", id));
    match fs::remove_file(&path) {
        Ok(()) => Ok(()),
        Err(e) if e.kind() == std::io::ErrorKind::NotFound => Ok(()),
        Err(e) => Err(StoreError::Io(e)),
    }
}

pub fn load_all_sessions() -> LoadResult {
    let root = tasks_root();
    let mut sessions = Vec::new();
    let mut toasts = Vec::new();
    let entries = match fs::read_dir(&root) {
        Ok(e) => e,
        Err(_) => return LoadResult { sessions, toasts },
    };
    for entry in entries.flatten() {
        let dir = entry.path();
        if !dir.is_dir() { continue; }
        let sid = entry.file_name().to_string_lossy().to_string();
        let mtime = entry.metadata().ok().and_then(|m| m.modified().ok()).unwrap_or(SystemTime::UNIX_EPOCH);
        let mut tasks = Vec::new();
        if let Ok(rd) = fs::read_dir(&dir) {
            for f in rd.flatten() {
                let name = f.file_name().to_string_lossy().to_string();
                if name.starts_with('.') { continue; }
                if name.ends_with(".tmp") { continue; }
                if !name.ends_with(".json") { continue; }
                let path = f.path();
                match fs::read_to_string(&path).and_then(|s| serde_json::from_str::<ClaudeTask>(&s).map_err(|e| std::io::Error::new(std::io::ErrorKind::InvalidData, e))) {
                    Ok(t) => tasks.push(t),
                    Err(_) => {
                        let short: String = sid.chars().take(8).collect();
                        toasts.push(format!("skipped {}/{} (parse error)", short, name));
                    }
                }
            }
        }
        if tasks.is_empty() { continue; }   // filter out zero-task sessions
        tasks.sort_by_key(|t| t.parse_id());
        sessions.push(LoadedSession { session_id: sid, mtime, tasks });
    }
    sessions.sort_by(|a, b| b.mtime.cmp(&a.mtime));
    LoadResult { sessions, toasts }
}

use crate::tasks::task::SessionId;
use std::collections::HashMap;
use std::path::PathBuf;
use std::sync::{Mutex, OnceLock};

fn projects_root() -> PathBuf {
    dirs::home_dir().unwrap().join(".claude/projects")
}

static CACHE: OnceLock<Mutex<Option<HashMap<SessionId, String>>>> = OnceLock::new();

fn cache() -> &'static Mutex<Option<HashMap<SessionId, String>>> {
    CACHE.get_or_init(|| Mutex::new(None))
}

fn build_cache() -> HashMap<SessionId, String> {
    let mut out = HashMap::new();
    let root = projects_root();
    let entries = match std::fs::read_dir(&root) {
        Ok(e) => e,
        Err(_) => return out,
    };
    for entry in entries.flatten() {
        let slug_dir = entry.path();
        if !slug_dir.is_dir() { continue; }
        let slug = entry.file_name().to_string_lossy().to_string();
        if let Ok(files) = std::fs::read_dir(&slug_dir) {
            for f in files.flatten() {
                let name = f.file_name().to_string_lossy().to_string();
                if !name.ends_with(".jsonl") { continue; }
                let sid = name.trim_end_matches(".jsonl").to_string();
                let label = label_from_jsonl(&f.path()).unwrap_or_else(|| slug.clone());
                out.insert(sid, label);
            }
        }
    }
    out
}

fn label_from_jsonl(path: &std::path::Path) -> Option<String> {
    use std::io::{BufRead, BufReader};
    let f = std::fs::File::open(path).ok()?;
    let reader = BufReader::new(f);
    for (i, line) in reader.lines().enumerate() {
        if i >= 20 { break; }
        let Ok(line) = line else { continue; };
        let Ok(val) = serde_json::from_str::<serde_json::Value>(&line) else { continue; };
        if let Some(cwd) = val.get("cwd").and_then(|v| v.as_str()) {
            let base = std::path::Path::new(cwd)
                .file_name()?
                .to_string_lossy()
                .to_string();
            if !base.is_empty() {
                return Some(base);
            }
        }
    }
    None
}

pub fn label_for(session_id: &SessionId) -> String {
    let mut guard = cache().lock().unwrap();
    if guard.is_none() {
        *guard = Some(build_cache());
    }
    if let Some(m) = guard.as_ref() {
        if let Some(lbl) = m.get(session_id) {
            return lbl.clone();
        }
    }
    // final fallback: 8-char prefix
    session_id.chars().take(8).collect()
}

pub fn refresh_labels() {
    let mut guard = cache().lock().unwrap();
    *guard = None;
}

pub fn is_blocked(task: &ClaudeTask, all_in_session: &[ClaudeTask]) -> Option<Vec<TaskId>> {
    if task.blocked_by.is_empty() { return None; }
    let open: Vec<TaskId> = task.blocked_by.iter()
        .filter(|bid| {
            all_in_session.iter()
                .find(|t| &t.id == *bid)
                .map(|t| t.status != Status::Completed)
                .unwrap_or(false)
        })
        .cloned()
        .collect();
    if open.is_empty() { None } else { Some(open) }
}

use crate::tasks::task::{ClaudeTask, SessionId, Status, TaskId};
use crate::tasks::store::{LockGuard, LoadResult, StoreError};
use crate::tasks::view::SessionGroup;
use crossterm::event::{KeyCode, KeyEvent, KeyModifiers};
use std::collections::{HashMap, HashSet};
use std::time::{Duration, Instant, SystemTime};

#[derive(Debug, Clone, Copy, Default)]
pub struct Focus {
    pub group: usize,
    pub task: Option<usize>,
}

#[derive(Debug, Clone)]
pub enum Filter {
    All,
    Session(SessionId),
    Subject(String),
}

#[derive(Debug)]
pub enum TaskAction {
    None,
    StatusCycled { id: TaskId, new: Status },
    Created { session: SessionId, id: TaskId },
    Deleted { id: TaskId },
    Toast(String),
    Quit,
}

pub struct TasksCore {
    pub groups: Vec<SessionGroup>,
    pub focus: Focus,
    pub expanded: HashSet<SessionId>,
    pub last_seen: HashMap<SessionId, SystemTime>,
    pub show_completed: bool,
    pub create_mode: Option<String>,
    pub pending_delete: Option<(SessionId, TaskId, Instant)>,
    pub last_toast: Option<(String, Instant)>,
    pub filter: Filter,
}

impl TasksCore {
    pub fn new() -> Self {
        let mut core = Self {
            groups: Vec::new(),
            focus: Focus::default(),
            expanded: HashSet::new(),
            last_seen: HashMap::new(),
            show_completed: false,
            create_mode: None,
            pending_delete: None,
            last_toast: None,
            filter: Filter::All,
        };
        core.full_reload();
        // expand the newest session by default
        if let Some(g) = core.groups.first() {
            core.expanded.insert(g.session_id.clone());
        }
        core
    }

    pub fn tick(&mut self) {
        // mtime-dirty reload; preserve focus by (sid, id)
        let focus_key = self.current_focus_key();
        self.full_reload();
        if let Some((sid, tid)) = focus_key {
            self.set_focus_by_key(&sid, tid.as_deref());
        }
        // prune expanded set
        let live: HashSet<SessionId> = self.groups.iter().map(|g| g.session_id.clone()).collect();
        self.expanded.retain(|s| live.contains(s));
        // clear stale pending_delete
        if let Some((_, _, t)) = self.pending_delete {
            if t.elapsed() > Duration::from_secs(2) {
                self.pending_delete = None;
            }
        }
    }

    fn full_reload(&mut self) {
        let LoadResult { sessions, toasts } = store::load_all_sessions();
        if let Some(t) = toasts.into_iter().next() {
            self.last_toast = Some((t, Instant::now()));
        }
        self.last_seen.clear();
        let mut groups = Vec::with_capacity(sessions.len());
        for s in sessions {
            self.last_seen.insert(s.session_id.clone(), s.mtime);
            let label = session::label_for(&s.session_id);
            groups.push(SessionGroup {
                session_id: s.session_id,
                label,
                mtime: s.mtime,
                tasks: s.tasks,
            });
        }
        self.groups = groups;
        // clamp focus
        if self.focus.group >= self.groups.len() {
            self.focus = Focus::default();
        }
    }

    fn current_focus_key(&self) -> Option<(SessionId, Option<TaskId>)> {
        let g = self.groups.get(self.focus.group)?;
        let tid = self.focus.task.and_then(|i| g.tasks.get(i).map(|t| t.id.clone()));
        Some((g.session_id.clone(), tid))
    }

    fn set_focus_by_key(&mut self, sid: &str, tid: Option<&str>) {
        let Some(gi) = self.groups.iter().position(|g| g.session_id == sid) else { return; };
        self.focus.group = gi;
        self.focus.task = tid.and_then(|id| {
            self.groups[gi].tasks.iter().position(|t| t.id == id)
        });
    }

    pub fn cycle_status(&mut self) -> TaskAction {
        let Some((sid, Some(_))) = self.current_focus_key() else { return TaskAction::None; };
        let g_idx = self.focus.group;
        let t_idx = self.focus.task.unwrap();
        let new_status = match self.groups[g_idx].tasks[t_idx].status {
            Status::Pending => Status::InProgress,
            Status::InProgress => Status::Completed,
            Status::Completed => Status::Pending,
        };
        let mut new_task = self.groups[g_idx].tasks[t_idx].clone();
        new_task.status = new_status.clone();
        let session_dir = store::tasks_root().join(&sid);
        match store::write_task(&session_dir, &new_task) {
            Ok(()) => {
                let id = new_task.id.clone();
                self.groups[g_idx].tasks[t_idx] = new_task;
                self.last_toast = Some((format!("#{} → {:?}", id, new_status), Instant::now()));
                TaskAction::StatusCycled { id, new: new_status }
            }
            Err(StoreError::LockTimeout) => {
                let id = self.groups[g_idx].tasks[t_idx].id.clone();
                self.last_toast = Some((format!("#{} locked, try again", id), Instant::now()));
                TaskAction::Toast(format!("#{} locked", id))
            }
            Err(e) => {
                self.last_toast = Some((format!("write failed: {:?}", e), Instant::now()));
                TaskAction::None
            }
        }
    }

    pub fn create_task(&mut self, subject: &str) -> TaskAction {
        if subject.trim().is_empty() { return TaskAction::None; }
        let Some(g) = self.groups.get(self.focus.group) else { return TaskAction::None; };
        let sid = g.session_id.clone();
        let next_id = g.tasks.iter().map(|t| t.parse_id()).max().map(|n| n.saturating_add(1)).unwrap_or(1);
        let task = ClaudeTask {
            id: next_id.to_string(),
            subject: subject.trim().to_string(),
            description: String::new(),
            active_form: String::new(),
            status: Status::Pending,
            blocks: vec![],
            blocked_by: vec![],
        };
        let dir = store::tasks_root().join(&sid);
        match store::write_task(&dir, &task) {
            Ok(()) => {
                let id = task.id.clone();
                // insert in numeric order
                if let Some(g) = self.groups.get_mut(self.focus.group) {
                    g.tasks.push(task);
                    g.tasks.sort_by_key(|t| t.parse_id());
                }
                self.last_toast = Some((format!("created #{}", id), Instant::now()));
                TaskAction::Created { session: sid, id }
            }
            Err(e) => {
                self.last_toast = Some((format!("create failed: {:?}", e), Instant::now()));
                TaskAction::None
            }
        }
    }

    pub fn arm_or_delete(&mut self) -> TaskAction {
        let Some((sid, Some(tid))) = self.current_focus_key() else { return TaskAction::None; };
        match self.pending_delete.take() {
            Some((psid, ptid, t)) if psid == sid && ptid == tid && t.elapsed() <= Duration::from_secs(2) => {
                let dir = store::tasks_root().join(&sid);
                match store::delete_task(&dir, &tid) {
                    Ok(()) => {
                        if let Some(g) = self.groups.get_mut(self.focus.group) {
                            g.tasks.retain(|t| t.id != tid);
                        }
                        self.last_toast = Some((format!("deleted #{}", tid), Instant::now()));
                        TaskAction::Deleted { id: tid }
                    }
                    Err(e) => {
                        self.last_toast = Some((format!("delete failed: {:?}", e), Instant::now()));
                        TaskAction::None
                    }
                }
            }
            _ => {
                self.pending_delete = Some((sid.clone(), tid.clone(), Instant::now()));
                self.last_toast = Some((format!("x again within 2s to delete #{}", tid), Instant::now()));
                TaskAction::Toast(format!("arm delete #{}", tid))
            }
        }
    }
}

impl Panel for TasksPanel {
    fn name(&self) -> &str { "tasks" }
    fn refresh_ms(&self) -> u64 { 2000 }
    fn tick(&mut self) { self.core.tick(); }

    fn handle_key(&mut self, key: KeyEvent) -> bool {
        match key.code {
            KeyCode::Char('j') | KeyCode::Down => { self.core.move_down(); true }
            KeyCode::Char('k') | KeyCode::Up   => { self.core.move_up(); true }
            KeyCode::Char('o') => { self.core.toggle_expand(); true }
            KeyCode::Char(' ') => { self.core.cycle_status(); true }
            _ => false,
        }
    }

        render_panel(f, area, &self.core);
    }

fn render_panel(f: &mut Frame, area: Rect, core: &TasksCore) {
    let width = WidthClass::from(area.width);
    let mut lines: Vec<Line> = Vec::new();
    for (gi, group) in core.groups.iter().enumerate() {
        let expanded = core.expanded.contains(&group.session_id);
        let glyph = if expanded { "▾" } else { "▸" };
        let header = match width {
            WidthClass::Tiny => format!("{} {}", glyph, group.label),
            _ => format!("{} {} · {} active", glyph, group.label, count_active(group)),
        };
        let header_style = if core.focus.group == gi && core.focus.task.is_none() {
            theme::pane_header_focused()
        } else {
            theme::dim()
        };
        lines.push(Line::from(Span::styled(header, header_style)));
        if expanded {
            for (ti, t) in group.tasks.iter().enumerate() {
                if !core.show_completed && matches!(t.status, crate::tasks::task::Status::Completed) { continue; }
                let glyph = match t.status {
                    crate::tasks::task::Status::Pending => "○",
                    crate::tasks::task::Status::InProgress => "◐",
                    crate::tasks::task::Status::Completed => "✓",
                };
                let blocked_marker = match (width, is_blocked(t, &group.tasks)) {
                    (WidthClass::Wide | WidthClass::Mid, Some(ids)) => {
                        format!("  ⛔#{}", ids.join(",#"))
                    }
                    _ => String::new(),
                };
                let row = format!("  {} #{}  {}{}", glyph, t.id, t.subject, blocked_marker);
                let style = if core.focus.group == gi && core.focus.task == Some(ti) {
                    theme::active_row()
                } else {
                    Style::default()
                };
                lines.push(Line::from(Span::styled(row, style)));
            }
        }
    }
    if lines.is_empty() {
        lines.push(Line::from(Span::styled("no tasks", theme::dim())));
    }
    let p = Paragraph::new(lines).block(Block::default().borders(Borders::ALL).title("tasks"));
    f.render_widget(p, area);
}
